143std::list<std::string> valideFreeClass(

145 int min = INT_MAX;
146 std::list<std::string> free_classes;
147
148 // first verify the class with the minimum number of students
149 for (auto &classe : it_count->second) {
150 if (classe.qtd < min) {
151 min = classe.qtd;

154 // then verify if the class is able to accept new students and add to the

156 for (auto &classe : it_count->second) {
